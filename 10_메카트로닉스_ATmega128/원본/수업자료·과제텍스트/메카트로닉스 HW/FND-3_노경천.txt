#include <avr/io.h>
#define F_CPU 16000000UL
#include <util/delay.h>
int main()
{
	char digit[10]={0x3f,0x06,0x5b,0x4f,0x66,
	                0x6d,0x7c,0x07,0x7f,0x67};
	int i, a, b, c;
	DDRC=0xff; //FND를연결한포트C를출력으로설정
	DDRG=0x0f; //FND선택신호인포트G를출력으로설정
	while(1) //반복적으로디스플레이
	{
		for(i=0; i<10; i++) //4-digit디스플레이를10번실행
		{
			for(a=0; a<10; a++) //디스플레이시간=10x10x20=2000ms
			{                       // 총 20초 소요 예정
				for(b=0; b<10; b++) //디스플레이시간=10x20=200ms
				{
					for(c=0; c<10; c++) //디스플레이시간=20ms
					{
							PORTC=digit[c];PORTG=0x01;_delay_ms(5);
							PORTC=digit[b];PORTG=0x02;_delay_ms(5);
							PORTC=digit[a];PORTG=0x04;_delay_ms(5);
							PORTC=0x00;_delay_ms(5);
					
					}
				}		
			}
		}
	}
}
