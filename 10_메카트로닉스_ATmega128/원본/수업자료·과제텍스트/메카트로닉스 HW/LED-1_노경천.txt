#include <avr/io.h>

void delay_sec(int sec)

{
	volatile int i,j,k;
	
	for (k=0; k<sec; k++)
	   for ( i=0; i<1000; i++)    // 내부 루프 1 번은 16클록 소요 가정
	      for  ( j=0; j<200; j++)  // 총 200 X 1000 x 16 = 약 0.2초
		  ;
}

int main(void)
{
  DDRA = 0xff;
  while(1)
  {
	  PORTA = 0xf0;
	  delay_sec(1);
	  PORTA = 0x0f;
	  delay_sec(1);
	  
	  
  }
}
