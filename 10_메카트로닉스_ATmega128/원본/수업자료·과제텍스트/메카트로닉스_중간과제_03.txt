#include <avr/io.h>
#include <stdlib.h>
#include <avr/interrupt.h>	// 인터럽트 처리 시 필요한 헤더 파일 include
#define F_CPU 16000000UL
#define __DELAY_BACKWARD_COMPATIBLE__
// Atmel Studio 7에서 _delay_ms( ) 함수에
// 정수 외 변수를 넣으려고 할 때 필요함
#include <util/delay.h>
// Motor Speed

#define STOP_SPEED 0 // Duty Cycle 0%의 값
#define LOW_SPEED 77 // duty cycle 30%의 값 (= 255 x 0.3)
#define MID_SPEED 153 // duty cycle 60%의 값 (= 255 x 0.6)
#define HIGH_SPEED 230 // duty cycle 90%의 값 (= 255 x 0.9)
// 모터 제어
// PB7 = PWMA, PB6 = AIN1, PB5 =AIN2, PB4 = STBY
#define MOTOR_CW 0xd0 // 모터 Forward : PWMA=1, AIN1=1, AIN2=0, STBY=1
#define MOTOR_CCW 0xb0 // 모터 Reverse : PWMA=1, AIN1=0, AIN2=1, STBY=1
#define MOTOR_BRAKE 0x70 // 모터 Short Brake : PWMA=0, AIN1=1, AIN2=1, STBY=1
#define MOTOR_STOP 0x90 // 모터 Stop : PWMA=1, AIN1=0, AIN2=0, STBY=1
#define MOTOR_STANDBY 0x00 // 모터 Standby : PWMA=0, AIN1=0, AIN2=0, STBY=0



#define GO_LOW 0
#define GO_MID 1
#define GO_HIGH 2
#define STOP 3

volatile int count=0, value=0;
volatile int state = STOP;      // 초기 상태는 IDLE


int motor_speed[4] = {STOP_SPEED, LOW_SPEED, MID_SPEED, HIGH_SPEED};
// STOP, LOW, MID, HIGH 4단계 임의 속도 선택


ISR(INT4_vect)
{
	_delay_ms(100);		        // 스위치 바운스 기간 동안 기다림
	if ((PINE & 0x10)==0x10)	// 인터럽트 입력 핀(PE4)을 다시 검사하여
	return;			            // 눌러진 상태가 아니면(‘1’) 인터럽트가 아니므로 리턴
	EIFR |= 1 << 4;      		// 눌러진 상태이면(‘0’) 그 사이에 바운스에 의하여 생긴
	// 인터럽트는 무효화
	state=GO_HIGH;
}
ISR (INT5_vect)
{
	_delay_ms(100);		        // 스위치 바운스 기간 동안 기다림
	if ((PINE & 0x20)==0x20)	// 인터럽트 입력 핀(PE5)을 다시 검사하여
	return;			            // 눌러진 상태가 아니면(‘1’) 인터럽트가 아니므로 리턴
	EIFR |= 1 << 5;		        // 눌러진 상태이면(‘0’) 그 사이에 바운스에 의하여 생긴
	// 인터럽트는 무효화
	state=GO_MID;
}

ISR (INT6_vect)
{
	_delay_ms(100);		        // 스위치 바운스 기간 동안 기다림
	if ((PINE & 0x40)==0x40)	// 인터럽트 입력 핀(PE6)을 다시 검사하여
	return;			            // 눌러진 상태가 아니면(‘1’) 인터럽트가 아니므로 리턴
	EIFR |= 1 << 6;		        // 눌러진 상태이면(‘0’) 그 사이에 바운스에 의하여 생긴
	// 인터럽트는 무효화
	state=GO_LOW;
}

ISR (INT7_vect)
{
	_delay_ms(100);	        	// 스위치 바운스 기간 동안 기다림
	if ((PINE & 0x80)==0x80)	// 인터럽트 입력 핀(PE7)을 다시 검사하여
	return;			            // 눌러진 상태가 아니면(‘1’) 인터럽트가 아니므로 리턴
	EIFR |= 1 << 7;	        	// 눌러진 상태이면(‘0’) 그 사이에 바운스에 의하여 생긴
	// 인터럽트는 무효화
	state = STOP;		    // 카운트 1씩 감소
}


int main(void)
{
	DDRA = 0xd0; // LED Display
	DDRB = 0xf0; // PB7 ~ PB4 사용
	PORTB = MOTOR_STANDBY; // 초기는 Standby 상태
	TCCR2 = 0x6b; // WGM10 = 11 : Fast PWM 모드
	// COM10 = 10 : match시 OC2 = 0,
	// TCNT 0x00에서 OC2 =1
	// CS2-0 = 011 : 64분주
	PORTB = MOTOR_CW; // 모터 기동 준비
	
	DDRE=0x00; //PE4(SW1),PE5(SW2)을포함한PE포트는입력신호
	EICRB=0xaa; // INT4, INT5,INT6, INT7 하강에지(FallingEdge)트리거
	EIMSK=0xf0; // INT4, INT5,INT6, INT7 인터럽트활성화
	SREG|=0x80; //sei();대신사용가능함

	while (1)
	{
		if (state==GO_HIGH)
		{OCR2 = motor_speed[HIGH_SPEED]; // 속도 선택, 정지, 저속, 중속, 고속 중 선택
			PORTA =0Xe0; // LED에 현재 속도 디스플레이
		}
		else if (state==GO_MID)
		{OCR2 = motor_speed[MID_SPEED]; // 속도 선택, 정지, 저속, 중속, 고속 중 선택
			PORTA =0XC0; // LED에 현재 속도 디스플레이
		}
		else if (state==GO_LOW)
		{OCR2 = motor_speed[LOW_SPEED]; // 속도 선택, 정지, 저속, 중속, 고속 중 선택
			PORTA =0X80; // LED에 현재 속도 디스플레이
		}
		else
		{OCR2 = motor_speed[STOP_SPEED]; // 속도 선택, 정지, 저속, 중속, 고속 중 선택
			PORTA =0X00; // LED에 현재 속도 디스플레이
		}
	}
	
}
