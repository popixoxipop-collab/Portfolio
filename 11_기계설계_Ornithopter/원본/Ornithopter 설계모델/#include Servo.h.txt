#include <Servo.h>

#define servoPin1 3
#define servoPin2 5
#define servoPin3 6
#define servoPin4 9
#define servoPin5 10

Servo servo1;
Servo servo2;
Servo servo3;
Servo servo4;
Servo servo5;

void servo1_OFF() {
  servo1.attach(servoPin1);
  for (int angle = 0; angle <= 90; angle++) { // 90도까지 회전
    servo1.write(angle);
    delay(10);
  }
  servo1.detach();
}

void servo1_ON() {
  servo1.attach(servoPin1);
  for (int angle = 90; angle >= 0; angle--) { // 0도로 돌아가기
    servo1.write(angle);
    delay(10);
  }
  servo1.detach();
}

void setup() {
  Serial.begin(115200);
  servo1.detach();
  servo2.detach();
  servo3.detach();
  servo4.detach();
  servo5.detach();
}

void loop() {
  delay(1000);
  servo1_OFF();
  delay(1000);
  servo1_ON();
  delay(1000);

  servo2_OFF();
  delay(1000);
  servo2_ON();
  delay(1000);

  servo3_OFF();
  delay(1000);
  servo3_ON();
  delay(1000);
